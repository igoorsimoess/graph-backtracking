% =====================================================================
%  estilo.tex  ->  cores, tipografia de titulos e espacamento
% =====================================================================

% ---------- Paleta consistente (usada tambem nas figuras) ------------
\definecolor{btDecisao}{HTML}{1F6FEB}  % azul   -> ponto de decisao
\definecolor{btValido}{HTML}{2E9E5B}   % verde  -> caminho valido
\definecolor{btPodado}{HTML}{D1453B}   % vermelho -> ramo podado/invalido
\definecolor{btNeutro}{HTML}{8A8F98}   % cinza  -> nao explorado
\definecolor{btFundo}{HTML}{F4F6F8}    % fundo suave das caixas
\definecolor{btTexto}{HTML}{1B1F24}    % texto base

% ---------- Espacamento ----------------------------------------------
\onehalfspacing
\setlength{\parindent}{0pt}
\setlength{\parskip}{0.6em}

% ---------- Titulos de secao (cara de artigo) ------------------------
\titleformat{\section}
  {\normalfont\Large\bfseries\color{btTexto}}
  {\thesection}{0.6em}{}
  [\vspace{2pt}\textcolor{btNeutro}{\titlerule[0.6pt]}]
\titleformat{\subsection}
  {\normalfont\large\bfseries\color{btTexto}}
  {\thesubsection}{0.5em}{}

\titlespacing*{\section}{0pt}{1.4em}{0.6em}
\titlespacing*{\subsection}{0pt}{1.0em}{0.4em}

% Cada seção começa em uma página nova (evita seções partidas entre folhas)
\newcommand{\sectionbreak}{\clearpage}

% ---------- Aparencia das legendas de figura -------------------------
\usepackage{caption}
\captionsetup{
  labelfont={bf,color=btDecisao},
  textfont={small},
  labelsep=endash,
  skip=6pt
}
